XONLINE_LATEST_ACCEPTED_GAMEINVITE

typedef struct _XONLINE_LATEST_ACCEPTED_GAMEINVITE{
    XUID xuidAcceptedUser;
    XUID xuidInvitingUser;
    CHAR szInvitingUserGamertag[XONLINE_GAMERTAG_SIZE];
    XNKID SessionID;
    FILETIME InviteAcceptTime;
    XUID xuidLogonUsers[XONLINE_MAX_LOGON_USERS];
} XONLINE_LATEST_ACCEPTED_GAMEINVITE, *PXONLINE_LATEST_ACCEPTED_GAMEINVITE;

Members

xuidAcceptedUser
An XUID structure uniquely identifying the user accepting the game invitation.
xuidInvitingUser
An XUID structure uniquely identifying the user making the game invitation.
szInvitingUserGamertag
Character string containing the Gamertag of the user making the game invitation.
SessionID
An XNKID structure representing the session.
InviteAcceptTime
A FILETIME structure containing the date and time of the game invitation acceptance.
xuidLogonUsers
Array of XUID structures that specify the users that were logged in at the time the cross-title invitation was accepted. Use the XOnlineGetUsers function to retrieve information for these users.
